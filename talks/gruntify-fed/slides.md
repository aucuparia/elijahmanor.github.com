```js
module.exports = function(grunt) {
  grunt.initConfig({
    pkg: grunt.file.readJSON('package.json'),
    concat: {
      options: {
        separator: ';'
      },
      dist: {
        src: ['src/**/*.js'],
        dest: 'dist/<%= pkg.name %>.js'
      }
    },
    uglify: {
      options: {
        banner:
      '/*!<%= pkg.name %> <%= grunt.template.today() %>*/\n'
      },
      dist: {
        files: {
          'dist/<%= pkg.name %>.min.js':
        ['<%= concat.dist.dest %>']
        }
      }
    },
    qunit: {
      files: ['test/**/*.html']
    },
    jshint: {
      files: ['Gruntfile.js', 'src/**/*.js', 'test/**/*.js'],
      options: {
        globals: {
          jQuery: true,
          console: true,
          module: true,
          document: true
        }
      }
    },
    watch: {
      files: ['<%= jshint.files %>'],
      tasks: ['jshint', 'qunit']
    }
  });

  grunt.loadNpmTasks('grunt-contrib-uglify');
  grunt.loadNpmTasks('grunt-contrib-jshint');
  grunt.loadNpmTasks('grunt-contrib-qunit');
  grunt.loadNpmTasks('grunt-contrib-watch');
  grunt.loadNpmTasks('grunt-contrib-concat');

  grunt.registerTask('test', ['jshint', 'qunit']);

  grunt.registerTask('default',
    ['jshint', 'qunit', 'concat', 'uglify']);
};
```

## [jit-grunt](https://github.com/shootaroo/jit-grunt) - Load time of Grunt does not slow down even if there are many plugins

### Gruntfile.js
```js
require('jit-grunt')(grunt);
```

### What about load-grunt-config?
```js
require('load-grunt-config')(grunt, {
  jitGrunt: true
});

```

# Run with Only Newer Files

## [grunt-newer](https://github.com/tschaub/grunt-newer) - Configure Grunt tasks to run with newer files only

### Gruntfile.js
```js
grunt.registerTask('lint', ['newer:jshint']);
```

### grunt/aliases.yaml
```yaml
lint:
  - 'newer:jshint'
```

# Custom Tasks

## Basic Tasks

```js
grunt.registerTask('hello', 'basic hi task', function(fn, ln) {
  if (arguments.length === 0) {
    grunt.log.writeln("Hello");
  } else {
    grunt.log.writeln("Hello " + fn + " " + ln);
  }
});
```

```shell
> grunt hello:elijah:manor
Hello Elijah Manor
```

-----



# Custom Tasks

## Multi Tasks

```js
grunt.registerMultiTask(
  'log',
  'multi log task',
  function() {
    grunt.log.writeln(
      this.target + ': ' +
    this.data
    );
});
```

---

```js
grunt.initConfig({
  log: {
    target1: [1, 2, 3],
    targer2: 'hello world'
  }
});
```

```output
> grunt log:target1
target1: 1,2,3

> grunt log
target1: 1,2,3
target2: hello world
```

# Custom Tasks

## `tasks` Folder

tasks/log.js
```js
module.exports = function(grunt) {
  grunt.registerMultiTask('log', 'multi log task', function() {
    grunt.log.writeln(this.target + ': ' + this.data);
  });
};
```
Load JavaScript files from `tasks` directory
```
grunt.loadTasks('tasks');
```

# Custom Plugin Template

Generate Grunt Plugin
```shell
npm install -g grunt-init
git clone git://github.com/gruntjs/grunt-init-gruntplugin.git ~/.grunt-init/gruntplugin
grunt-init gruntplugin
npm install
```

# Custom Plugin Structure

## `grunt-init gruntplugin`

```
.
├── Gruntfile.js
├── LICENSE-MIT
├── README.md
├── package.json
├── tasks
│   └── your-plugin.js
└── test
    ├── expected
    ├── fixtures
    └── your-plugin_test.js
```


Publish to `npm`
```
npm publish
```
