      options: {

    qunit: {
      files: ['test/**/*.html']
    },
    jshint: {
      files: ['Gruntfile.js', 'src/**/*.js', 'test/**/*.js'],
      options: {
        globals: {
          jQuery: true,
          console: true,
          module: true,
          document: true
        }
      }
    },
    watch: {
      files: ['<%= jshint.files %>'],
      tasks: ['jshint', 'qunit']
    }
  });

  grunt.loadNpmTasks('grunt-contrib-uglify');
  grunt.loadNpmTasks('grunt-contrib-jshint');
  grunt.loadNpmTasks('grunt-contrib-qunit');
  grunt.loadNpmTasks('grunt-contrib-watch');
  grunt.loadNpmTasks('grunt-contrib-concat');

  grunt.registerTask('test', ['jshint', 'qunit']);

  grunt.registerTask('default', 
    ['jshint', 'qunit', 'concat', 'uglify']);
};

│   ├── Gruntfile.js
│   ├── grunt

│   │   ├── plugin1.js
│   │   ├── plugin2.js

# Custom Tasks

## `tasks` Folder

tasks/log.js
```js
module.exports = function(grunt) {
  grunt.registerMultiTask('log', 'multi log task', function() {
    grunt.log.writeln(this.target + ': ' + this.data);
  });
};
```
Load JavaScript files from `tasks` directory
```
grunt.loadTasks('tasks');
```

# Custom Plugin Template

Generate Grunt Plugin
```shell
npm install -g grunt-init
git clone git://github.com/gruntjs/grunt-init-gruntplugin.git ~/.grunt-init/gruntplugin
grunt-init gruntplugin
npm install
```

# Custom Plugin Structure

## `grunt-init gruntplugin`

```
.
├── Gruntfile.js
├── LICENSE-MIT
├── README.md
├── package.json
├── tasks
│   └── your-plugin.js
└── test
    ├── expected
    ├── fixtures
    └── your-plugin_test.js
```


Publish to `npm`
```
npm publish
```

# Grunt.next Goals

## [The State of Grunt August 2014](https://cowboy.github.io/state-of-grunt-fe-summit-2014-talk/#55)

* More modular and maintainable
* Support existing 0.4.x plugins via optional adapter, & support Gulp & Broccoli plugins
* Tasks support stream-based and promise-based input/output
* Support tasks & pipelines that can run in series or parallel
* All Grunt tasks and file I/O will be async (under the hood)
